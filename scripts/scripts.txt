# modules #
module load chpc/BIOMODULES
module load minimap2/2.24

# Variables #
raw=/mnt/lustre/users/ksekane/raw/
work=/mnt/lustre/users/ksekane/mapping/

# align data #
cd ${work}
minimap2 -ax map-hifi ${work}/PPantherinum_sequence.fasta ${raw}/*.fastq.gz > mapping1.sam
